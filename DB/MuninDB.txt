CREATE TABLE [dbo].[CourseProgress]
(
	[Id] INT NOT NULL PRIMARY KEY IDENTITY (1, 1),
	[CourseId] INT NULL, 
    [UserId] INT NULL, 
    [FinishedId] BIT NOT NULL DEFAULT 0, 
    CONSTRAINT [FK_CourseProgress_Courses] FOREIGN KEY ([CourseId]) REFERENCES [Courses]([Id]),
    CONSTRAINT [FK_CourseProgress_Users] FOREIGN KEY ([UserId]) REFERENCES [Users]([Id]),

	
	
)
CREATE TABLE [dbo].[Courses]
(
	[Id] INT NOT NULL PRIMARY KEY IDENTITY (1, 1), 
    [Name] NVARCHAR(50) NULL, 
    [Description] NVARCHAR(MAX) NULL,
	
)
CREATE TABLE [dbo].[Users]
(
	[Id] INT NOT NULL PRIMARY KEY IDENTITY (1, 1),
	[Name] NVARCHAR(50) NOT NULL, 
    [Password] NVARCHAR(50) NOT NULL


)
